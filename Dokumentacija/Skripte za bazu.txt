CREATE TABLE "Delatnost" (
    "delatnostID" text,
    naziv text,
    opis text,
    tip text,           --mozda je ovo posebna tabela jer izgleda da nema add constraint ... check(tip in ('problemi', 'iskustva'...))
    PRIMARY KEY ("delatnostID")
    );
    
CREATE TABLE "Korisnik" (
    "korisnikID" text,
    ime text,
    prezime text,
    email text,
    telefon text,
    vebsajt text,
    PRIMARY KEY ("korisnikID")
    );--kod njega dinamicki da se dodaju kolone za delatnosti
    
CREATE TABLE "Tema" (
    "delatnostID" text,
    "korisnikID" text,
    "temaID" text,
    sadrzaj text,
    vidljivost text, --mozda i kao tabela da bi bio partition key?
    datumKreiranja text,
    slika text, --ili da moze da ima vise slika?
    PRIMARY KEY ("delatnostID", "temaID") --partition key moze da bude i korisnikID
    );
